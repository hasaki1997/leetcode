```js
const countBits = function (num) {
  let res = [], count = 0, str = '';
  for (let i = 0; i <= num; ++i) {
    let n = i;
    while (n !== 0) {
      if (res[n]) {
        count += res[n];
        break;
      }
      if (n % 2 === 1) {
        ++count;
      }
      n = parseInt(n / 2);
    }
    res[i] = count;
    count = 0;
  }
  return res;
};
```
